: add-advance  \ ( a1 a2 a3 -- a1+ a2+ a3+ )
	\ "add a2 and a3, store value in a1. Advance all three addresses."
	dup >r -rot dup >r -rot dup >r -rot \ ( a1 a2 a3 -|- a1 a2 a3 )
	@ swap @ +                          \ ( a1 a2+a3 -|- a1 a2 a3 )
	swap !                              \ ( nil -|- a1 a2 a3      )
	r> r> r>					        \ ( a1 a2 a3    --        )
	cell + rot cell + rot cell + rot    \ ( a1+ a2+ a3+ --        )
	;
